# Full-MC system example: absolute planar sensitivity, GE MEGP (hexagonal-hole Pb) + NaI at 10 cm.
#
# Selects the GE MEGP preset with a single IncludeFile, then adds the world, physics, source, scorer,
# detector extents and placements. The collimator is sized wider than the crystal so no photons bypass
# its edge. Reproduces the validated result (9.32e-5 counts/decay; docs/validation.md section 4).
#
# In post-processing, count events whose deposited energy falls in the Tc-99m photopeak window
# (126.5-154.6 keV) and divide by NumberOfHistoriesInRun to get counts/decay.
#
# Run from this example's folder so the relative include resolves:
#   topas ge_megp_sensitivity.txt

IncludeFile = ../../systems/ge_megp.txt      # Ma/Lead, Ma/NaI, Ge/Collim, Ge/Crystal (spec)

i:Ts/NumberOfThreads = 0
b:Ge/QuitIfOverlapDetected = "True"

sv:Ph/Default/Modules     = 1 "g4em-penelope"
b:Ph/Default/Fluorescence = "True"
d:Ph/Default/CutForGamma    = 0.05 mm
d:Ph/Default/CutForElectron = 0.05 mm

d:Ge/World/HLX = 250. mm
d:Ge/World/HLY = 250. mm
d:Ge/World/HLZ = 200. mm
s:Ge/World/Material = "Vacuum"

# Detector extents + placement (the spec-fixed hole/septum/length/material come from the preset).
# Collimator front face at z = 0; NHoles chosen so the collimator exceeds the crystal footprint.
i:Ge/Collim/NHolesX = 50
i:Ge/Collim/NHolesY = 50
d:Ge/Collim/TransZ  = 29.0 mm

d:Ge/Crystal/HLX    = 90. mm
d:Ge/Crystal/HLY    = 90. mm
d:Ge/Crystal/TransZ = 63.0 mm

# Isotropic point source at 10 cm in front of the collimator
s:Ge/Src/Type   = "TsBox"
s:Ge/Src/Parent = "World"
s:Ge/Src/Material = "Vacuum"
d:Ge/Src/HLX = 0.01 mm
d:Ge/Src/HLY = 0.01 mm
d:Ge/Src/HLZ = 0.01 mm
d:Ge/Src/TransZ = -100. mm

s:So/Beam/Type = "Isotropic"
s:So/Beam/Component = "Src"
s:So/Beam/BeamParticle = "gamma"
d:So/Beam/BeamEnergy = 140.5 keV
i:So/Beam/NumberOfHistoriesInRun = 50000000

s:Sc/Spec/Quantity   = "EDepSpectrum"
s:Sc/Spec/Component  = "Crystal"
s:Sc/Spec/OutputType = "ASCII"
s:Sc/Spec/OutputFile = "sensitivity_ge_megp_140p5keV_10cm"
s:Sc/Spec/IfOutputFileAlreadyExists = "Overwrite"
b:Ts/PauseBeforeQuit = "False"
