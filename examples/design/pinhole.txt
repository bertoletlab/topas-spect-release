# Design example: single pinhole collimator.
# TsPinholeCollimator = an absorber plate with a knife-edge double-cone aperture. Keep the plate
# large enough to cover the detector field of view (else photons bypass the plate). The pinhole
# projects an inverted, magnified image (m = l / h).
# Run from this example's folder: topas pinhole.txt

i:Ts/NumberOfThreads = 1
b:Ge/QuitIfOverlapDetected = "True"

sv:Ma/Lead/Components = 1 "Lead"
uv:Ma/Lead/Fractions  = 1 1.0
d:Ma/Lead/Density     = 11.35 g/cm3
sv:Ma/NaI/Components = 2 "Sodium" "Iodine"
uv:Ma/NaI/Fractions  = 2 0.1534 0.8466
d:Ma/NaI/Density     = 3.67 g/cm3

sv:Ph/Default/Modules = 1 "g4em-penelope"
b:Ph/Default/Fluorescence = "True"

d:Ge/World/HLX = 300. mm
d:Ge/World/HLY = 300. mm
d:Ge/World/HLZ = 400. mm
s:Ge/World/Material = "Vacuum"

s:Ge/Collim/Type           = "TsPinholeCollimator"
s:Ge/Collim/Parent         = "World"
s:Ge/Collim/Material       = "Lead"
d:Ge/Collim/PlateThickness  = 7.0 mm
d:Ge/Collim/PinholeDiameter = 2.0 mm
d:Ge/Collim/AcceptanceAngle = 90. deg
d:Ge/Collim/PlateHLX = 150. mm          # large plate: only the pinhole transmits (no bypass)
d:Ge/Collim/PlateHLY = 150. mm
d:Ge/Collim/TransZ = 0. mm              # pinhole mid-plane at z = 0

s:Ge/Crystal/Type     = "TsBox"
s:Ge/Crystal/Parent   = "World"
s:Ge/Crystal/Material = "NaI"
d:Ge/Crystal/HLX = 120. mm
d:Ge/Crystal/HLY = 120. mm
d:Ge/Crystal/HLZ = 4.75 mm
d:Ge/Crystal/TransZ = 100. mm           # detector at l = 100 mm behind the pinhole

s:Ge/Src/Type = "TsBox"
s:Ge/Src/Parent = "World"
s:Ge/Src/Material = "Vacuum"
d:Ge/Src/HLX = 0.01 mm
d:Ge/Src/HLY = 0.01 mm
d:Ge/Src/HLZ = 0.01 mm
d:Ge/Src/TransZ = -50. mm               # object at h = 50 mm in front -> magnification m = l/h = 2
s:So/Beam/Type = "Isotropic"
s:So/Beam/Component = "Src"
s:So/Beam/BeamParticle = "gamma"
d:So/Beam/BeamEnergy = 140.5 keV
i:So/Beam/NumberOfHistoriesInRun = 5000000

s:Sc/Spec/Quantity   = "EDepSpectrum"
s:Sc/Spec/Component  = "Crystal"
s:Sc/Spec/OutputType = "ASCII"
s:Sc/Spec/OutputFile = "pinhole"
s:Sc/Spec/IfOutputFileAlreadyExists = "Overwrite"
b:Ts/PauseBeforeQuit = "False"
